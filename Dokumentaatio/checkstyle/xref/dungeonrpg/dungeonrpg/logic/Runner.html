1   package dungeonrpg.dungeonrpg.logic;
2   
3   import dungeonrpg.dungeonrpg.entities.Enemy;
4   import dungeonrpg.dungeonrpg.entities.Loot;
5   import dungeonrpg.dungeonrpg.ui.GameIF;
6   import java.util.HashMap;
7   
8   /**
9    * Class that runs the game logic methods.
10   */
11  public class Runner {
12  
13      private Logic l;
14      private GameIF ui;
15  
16      public Runner(GameIF ui, Logic l) {
17          this.l = l;
18          this.ui = ui;
19      }
20  
21      public Logic getLogic() {
22          return l;
23      }
24  
25      /**
26       * Method that runs the game methods in order.
27       *
28       * @return boolean for testing purposes
29       */
30      public boolean runGame() {
31          boolean ko = false;
32          if (ui.strt()) {
33              while (!ko) {
34                  ko = battle(l.getWrld().checkForMonsters(l.getPlr().getPosX(), l.getPlr().getPosY()));
35                  if (ko) {
36                      break;
37                  }
38                  loot(l.getWrld().checkForLoot(l.getPlr().getPosX(), l.getPlr().getPosY()));
39                  ko = move(ui.whereTo());
40              }
41          }
42          return ko;
43      }
44  
45      /**
46       * Method that runs the battle portion of gameplay.
47       *
48       * @param enmy opponent
49       * @return true if player dies, false otherwise
50       */
51      public boolean battle(Enemy enmy) {
52          if (enmy != null) {
53              ui.alert(6, enmy.getName(), 0);
54              while (true) {
55                  int n = l.getPlr().getTurnpoints();
56  
57                  l.getPlr().unParry();
58                  int k = 0;
59                  while (n > 0) {
60                      ui.alert(16, null, n);
61                      k = l.plrTurn(enmy, ui.battle());
62                      ui.alert(k, enmy.getName(), 0);
63                      if (k == 4) {
64                          break;
65                      }
66                      n--;
67                  }
68                  if (k == 4) {
69                      break;
70                  }
71                  if (enmyTurn(enmy)) {
72                      return true;
73                  }
74              }
75          }
76          return false;
77      }
78  
79      /**
80       * Adds Loot objects to inventory if there are any in current position.
81       *
82       * @param loot item to be added
83       */
84      public void loot(Loot loot) {
85          if (loot != null) {
86              l.getPlr().addLoot(loot);
87              loot.throwOffEdge(l.getWrld().getSize());
88              ui.alert(14, loot.getName(), 0);
89          }
90      }
91  
92      /**
93       * Movement portion of gameplay.
94       *
95       * @param d option that the user has chosen. 1,2,3,4 for movement,
96       * 5=inventory, 6=quit.
97       * @return true if d=6, false otherwise.
98       */
99      public boolean move(int d) {
100         while (true) {
101             if (d >= 1 && d <= 4) {
102                 if (l.movement(d)) {
103                     break;
104                 }
105                 ui.alert(1, null, 0);
106             } else if (d == 6) {
107                 return true;
108             } else if (d == 5) {
109                 inventoryCheck(l.getPlr().getLoot());
110             }
111             d = ui.whereTo();
112         }
113         return false;
114     }
115 
116     /**
117      * Method for choosing Loot objects from inventory.
118      *
119      * @param loot HashMap of items in player's inventory.
120      */
121     public void inventoryCheck(HashMap<String, Loot> loot) {
122         if (loot.isEmpty()) {
123             ui.alert(15, null, 0);
124         } else {
125             String[] items = ui.inventory(loot.keySet());
126             String i1 = items[0];
127             String i2 = items[1];
128             l.getPlr().equip(loot.get(i1), loot.get(i2));
129         }
130     }
131 
132     /**
133      * Method for enemy's "AI".
134      *
135      * @param enmy
136      * @return true if player dies and false otherwise.
137      */
138     public boolean enmyTurn(Enemy enmy) {
139         int n = enmy.getTurnpoints();
140         while (n > 0) {
141             int k = l.enmyActn(enmy);
142             if (k == 1 || k == 2) {
143                 ui.alert(11, enmy.getName(), 0);
144                 if (k == 1) {
145                     if (l.getPlr().getHealth() == 0) {
146                         ui.alert(8, enmy.getName(), 0);
147                         return true;
148                     } else {
149                         ui.alert(9, enmy.getName(), 0);
150                     }
151                 } else {
152                     ui.alert(12, null, 0);
153                 }
154             }
155             if (k == 3) {
156                 ui.alert(13, enmy.getName(), 0);
157                 ui.alert(10, enmy.getName(), 0);
158             }
159             if (k == 4) {
160                 ui.alert(13, enmy.getName(), 0);
161                 ui.alert(12, enmy.getName(), 0);
162             }
163             n--;
164         }
165         enmy.unParry();
166         return false;
167     }
168 }
