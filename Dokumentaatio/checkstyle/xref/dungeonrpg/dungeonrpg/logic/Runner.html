6   package dungeonrpg.dungeonrpg.logic;

8   import dungeonrpg.dungeonrpg.entities.Enemy;
9   import dungeonrpg.dungeonrpg.ui.Interface;
10  import java.util.concurrent.TimeUnit;
11  
12  /**
13   *
14   * @author Jonne
15   */
16  public class Runner {
17  
18      private Logic l;
19      private Interface ui;
20  
21      public Runner() {
22          this.l = new Logic();
23          this.ui = new Interface();
24      }
25  
26      public void runGame() {
27          if (ui.strt()) {
28              boolean ko = false;
29              while (!ko) {
30                  Enemy enmy = l.getWrld().checkForMonsters(l.getPlr().getPosX(), l.getPlr().getPosY());
31                  if (enmy != null) {
32                      ui.alert(6, enmy.getName());
33                      while (true) {
34                          int n = l.getPlr().getTurnpoints();
35                          l.getPlr().unParry();
36                          int k = 0;
37                          while (n > 0) {
38                              k = l.plrTurn(enmy, ui.battle());
39                              ui.alert(k, enmy.getName());
40                              if (k == 4) {
41                                  break;
42                              }
43                              n--;
44                          }
45                          if (k == 4) {
46                              break;
47                          }
48                          if (enmyTurn(enmy)) {
49                              ko = true;
50                              break;
51                          }
52                      }
53                  }
54                  if (ko) {
55                      break;
56                  }
57                  int d = ui.whereTo();
58                  while (true) {
59                      if (d == 5) {
60                          ko = true;
61                          break;
62                      }
63                      if (l.movement(d)) {
64                          break;
65                      }
66                      ui.alert(1, null);
67                      d = ui.whereTo();
68                  }
69              }
70          }
71      }
72  
73      public boolean enmyTurn(Enemy enmy) {
74          int n = enmy.getTurnpoints();
75          while (n > 0) {
76              int k = l.enmyActn(enmy);
77              if (k == 1 || k == 2) {
78                  ui.alert(11, enmy.getName());
79                  if (k == 1) {
80                      if (l.getPlr().getHealth() == 0) {
81                          ui.alert(8, enmy.getName());
82                          return true;
83                      } else {
84                          ui.alert(9, enmy.getName());
85                      }
86                  } else {
87                      ui.alert(12, null);
88                  }
89              }
90              if (k == 3) {
91                  ui.alert(13, enmy.getName());
92                  ui.alert(10, enmy.getName());
93              }
94              if (k == 4) {
95                  ui.alert(13, enmy.getName());
96                  ui.alert(12, enmy.getName());
97              }
98              n--;
99          }
100         enmy.unParry();
101         return false;
102     }
103 
104     public void waitForABit(int n) {
105         try {
106             TimeUnit.MILLISECONDS.sleep(n);
107         } catch (InterruptedException e) {
108         }
109     }
110 }
